    **CRITICAL: Nested config classes MUST extend BaseModel, NOT BaseSettings.** Only the root `StakeSettings` class should be `BaseSettings`. This is required for `env_nested_delimiter="__"` to work correctly — if nested classes are also `BaseSettings`, they each try to load their own env vars independently instead of delegating to the parent's prefix/delimiter logic. With `BaseModel`, `STAKE_PARSER__MODEL` will correctly populate `StakeSettings().parser.model`.

    ```python
    from pydantic import BaseModel, Field
    from pydantic_settings import BaseSettings, SettingsConfigDict
    from functools import lru_cache

    class ParserSettings(BaseModel):
        model: str = Field(default="google/gemini-2.0-flash-001", description="OpenRouter model for parse step")
        temperature: float = Field(default=0.0, description="LLM temperature for parsing")
        max_tokens: int = Field(default=4000, description="Max tokens for parse response")

    class RedisSettings(BaseModel):
        url: str = Field(default="redis://redis:6379", description="Redis URL for FSM storage")
        state_ttl: int = Field(default=86400, description="FSM state TTL in seconds (24h)")
        data_ttl: int = Field(default=86400, description="FSM data TTL in seconds (24h)")

    class BankrollSettings(BaseModel):
        default_stake_pct: float = Field(default=0.02, description="Default stake as fraction of bankroll")

    class AuditSettings(BaseModel):
        log_path: str = Field(default="data/stake_audit.jsonl", description="Path to JSONL audit log")

    class StakeSettings(BaseSettings):
        model_config = SettingsConfigDict(
            env_prefix="STAKE_",
            env_nested_delimiter="__",
            env_file=".env",
            extra="ignore"
        )
        parser: ParserSettings = Field(default_factory=ParserSettings)
        redis: RedisSettings = Field(default_factory=RedisSettings)
        bankroll: BankrollSettings = Field(default_factory=BankrollSettings)
        audit: AuditSettings = Field(default_factory=AuditSettings)
        database_path: str = Field(default="races.db", description="SQLite database path")
        openrouter_api_key: str = Field(default="", description="OpenRouter API key")
        telegram_bot_token: str = Field(default="", description="Telegram bot token")
        telegram_chat_id: str = Field(default="", description="Telegram chat ID")

    @lru_cache()
    def get_stake_settings() -> StakeSettings:
        return StakeSettings()
    ```

    ```python
    import sqlite3

    def run_stake_migrations(db_path: str = "races.db") -> None:
        conn = sqlite3.connect(db_path)
        cursor = conn.cursor()

        cursor.execute("""
            CREATE TABLE IF NOT EXISTS stake_bankroll (
                id INTEGER PRIMARY KEY CHECK (id = 1),
                balance_usdt REAL NOT NULL,
                stake_pct REAL NOT NULL DEFAULT 0.02,
                updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)

        cursor.execute("""
            CREATE TABLE IF NOT EXISTS stake_pipeline_runs (
                run_id INTEGER PRIMARY KEY AUTOINCREMENT,
                raw_input TEXT NOT NULL,
                parsed_race_json TEXT,
                user_confirmed INTEGER DEFAULT 0,
                user_changes_json TEXT,
                bankroll_at_run REAL,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)

        conn.commit()
        conn.close()
    ```

    The `CHECK (id = 1)` constraint ensures only one bankroll row exists (singleton pattern).

    **Step 2: Create repository in `services/stake/bankroll/repository.py`**
    Follow the Repository pattern from `src/database/repositories.py`:

    ```python
    import sqlite3
    from typing import Optional
    from services.stake.bankroll.migrations import run_stake_migrations

    class BankrollRepository:
        def __init__(self, db_path: str = "races.db"):
            self.db_path = db_path
            run_stake_migrations(db_path)

        def get_balance(self) -> Optional[float]:
            conn = sqlite3.connect(self.db_path)
            cursor = conn.cursor()
            cursor.execute("SELECT balance_usdt FROM stake_bankroll WHERE id = 1")
            row = cursor.fetchone()
            conn.close()
            return float(row[0]) if row else None

        def set_balance(self, balance: float) -> None:
            conn = sqlite3.connect(self.db_path)
            cursor = conn.cursor()
            cursor.execute("""
                INSERT INTO stake_bankroll (id, balance_usdt, stake_pct)
                VALUES (1, ?, 0.02)
                ON CONFLICT(id) DO UPDATE SET
                    balance_usdt = excluded.balance_usdt,
                    updated_at = CURRENT_TIMESTAMP
            """, (balance,))
            conn.commit()
            conn.close()

        def get_stake_pct(self) -> float:
            conn = sqlite3.connect(self.db_path)
            cursor = conn.cursor()
            cursor.execute("SELECT stake_pct FROM stake_bankroll WHERE id = 1")
            row = cursor.fetchone()
            conn.close()
            return float(row[0]) if row else 0.02

        def set_stake_pct(self, pct: float) -> None:
            conn = sqlite3.connect(self.db_path)
            cursor = conn.cursor()
            cursor.execute("""
                INSERT INTO stake_bankroll (id, balance_usdt, stake_pct)
                VALUES (1, 0.0, ?)
                ON CONFLICT(id) DO UPDATE SET
                    stake_pct = excluded.stake_pct,
                    updated_at = CURRENT_TIMESTAMP
            """, (pct,))
            conn.commit()
            conn.close()
    ```
