    **Step 2: Create `services/stake/pipeline/nodes.py`**
    Pipeline node functions. Each node takes PipelineState and returns partial update dict:

    - `parse_node(state)`: Creates StakeParser, calls `await parser.parse(state["raw_input"])`, returns `{"parsed_race": result, "detected_bankroll": result.detected_bankroll}`. Catches exceptions and returns `{"error": str(e)}`. After parsing, checks for ambiguity: if `runner_count` is set but doesn't match `len(runners)`, or if more than 30% of runners are missing `win_odds`, or if `track` is None, adds those field names to `ambiguous_fields` list. This drives the PIPELINE-02 clarifying question flow.

    - `calc_node(state)`: Takes `state["parsed_race"]`, iterates runners. For each active runner with `win_odds` and `win_odds_format`, calls `to_decimal()` to get decimal odds, then `implied_probability()`. Builds `enriched_runners` list of dicts: `{"number": r.number, "name": r.name, "decimal_odds": ..., "implied_prob": ..., "odds_drift": ..., "status": r.status, ...all other runner fields}`. Calculates `overround_raw` from all runners with odds, `overround_active` from active runners only via `recalculate_without_scratches`. Returns `{"enriched_runners": ..., "overround_raw": ..., "overround_active": ...}`.

    ```python
    from langgraph.graph import StateGraph, END
    from services.stake.pipeline.state import PipelineState
    from services.stake.pipeline.nodes import parse_node, calc_node

    def error_router(state: PipelineState) -> str:
        if state.get("error"):
            return "error"
        return "continue"

    def build_pipeline_graph():
        graph = StateGraph(PipelineState)
        graph.add_node("parse", parse_node)
        graph.add_node("calc", calc_node)
        graph.add_conditional_edges("parse", error_router, {
            "error": END,
            "continue": "calc"
        })
        graph.add_edge("calc", END)
        graph.set_entry_point("parse")
        return graph.compile()
    ```

    ```python
    from services.stake.pipeline.state import PipelineState

    def format_race_summary(state: PipelineState) -> str:
        """Format parsed race as HTML for Telegram message. Per PARSE-04."""
        race = state.get("parsed_race")
        if not race:
            return "No race data parsed."

        lines = []
        # Race header
        header_parts = []
        if race.track: header_parts.append(f"<b>{race.track}</b>")
        if race.race_number: header_parts.append(f"Race {race.race_number}")
        if race.race_name: header_parts.append(race.race_name)
        lines.append(" | ".join(header_parts) if header_parts else "<b>Race Summary</b>")

        # Race details
        details = []
        if race.distance: details.append(f"Distance: {race.distance}")
        if race.surface: details.append(f"Surface: {race.surface}")
        if race.place_terms: details.append(f"Place: {race.place_terms}")
        if details:
            lines.append("  ".join(details))

        # Overround
        ovr_raw = state.get("overround_raw")
        ovr_active = state.get("overround_active")
        if ovr_raw is not None:
            margin = (ovr_raw - 1) * 100
            lines.append(f"\nOverround: {ovr_raw:.2f} (margin {margin:.1f}%)")
        if ovr_active is not None and ovr_active != ovr_raw:
            margin_a = (ovr_active - 1) * 100
            lines.append(f"Active only: {ovr_active:.2f} (margin {margin_a:.1f}%)")

        # Runners table
        lines.append("\n<b>Runners:</b>")
        enriched = state.get("enriched_runners", [])
        for r in enriched:
            status_icon = "SCRATCHED" if r.get("status") == "scratched" else ""
            odds_str = f"{r.get('decimal_odds', '?'):.2f}" if r.get('decimal_odds') else "—"
            prob_str = f"{r.get('implied_prob', 0)*100:.1f}%" if r.get('implied_prob') else "—"
            tags_str = f" [{', '.join(r['tags'])}]" if r.get('tags') else ""
            drift_str = ""
            if r.get('odds_drift') is not None:
                drift_str = f" ({r['odds_drift']:+.1f}%)"

            if status_icon:
                lines.append(f"  <s>{r['number']}. {r['name']}</s> SCRATCHED")
            else:
                lines.append(f"  {r['number']}. {r['name']} — {odds_str} ({prob_str}){drift_str}{tags_str}")

        # Bet types
        if race.bet_types_available:
            lines.append(f"\nBet types: {', '.join(race.bet_types_available)}")

        # Ambiguous fields warning (PIPELINE-02)
        ambiguous = state.get("ambiguous_fields", [])
        if ambiguous:
            lines.append(f"\n<i>Note: Some data may be incomplete ({', '.join(ambiguous)}). Please review carefully.</i>")

        return "\n".join(lines)
    ```

    **Step 1: Create `services/stake/handlers/pipeline.py`**
    Handles text paste (INPUT-01) and .txt document (INPUT-02) in idle state. Per PIPELINE-01 (progressive updates), PIPELINE-02 (clarifying question for ambiguous data), PIPELINE-05 (duplicate warning):

    ```python
    import io
    import sys
    from pathlib import Path
    sys.path.insert(0, str(Path(__file__).parent.parent.parent.parent))

    from aiogram import Router, F, Bot
    from aiogram.types import Message
    from aiogram.fsm.context import FSMContext

    from services.stake.states import PipelineStates
    from services.stake.settings import get_stake_settings
    from services.stake.handlers.commands import balance_header
    from services.stake.pipeline.graph import build_pipeline_graph
    from services.stake.pipeline.formatter import format_race_summary
    from services.stake.keyboards.stake_kb import confirm_parse_kb, bankroll_confirm_kb, bankroll_input_kb
    from services.stake.bankroll.repository import BankrollRepository
    from services.stake.audit.logger import AuditLogger

    router = Router(name="pipeline")

    async def _run_parse_pipeline(message: Message, state: FSMContext, raw_text: str):
        """Common pipeline logic for both paste and document input."""
        settings = get_stake_settings()
        header = balance_header(settings.database_path)
        audit = AuditLogger()

        # PIPELINE-05: check for active pipeline
        current_state = await state.get_state()
        if current_state and current_state != PipelineStates.idle.state:
            await message.answer(
                f"{header}"
                "A pipeline is already active. Use /cancel first, "
                "or wait for it to complete."
            )
            return

        # Set state to parsing
        await state.set_state(PipelineStates.parsing)
        await state.update_data(raw_input=raw_text)

        # AUDIT-01: log pipeline start
        audit.log_entry("pipeline_start", {"raw_input": raw_text[:500]})

        # PIPELINE-01: progressive update — parsing step
        progress_msg = await message.answer(f"{header}Parsing race data...")

        # Run LangGraph pipeline
        try:
            graph = build_pipeline_graph()
            result = await graph.ainvoke({"raw_input": raw_text})
        except Exception as e:
            await state.set_state(PipelineStates.idle)
            await message.answer(f"{header}Parse error: {str(e)}\n\nPlease try again.")
            audit.log_entry("parse_error", {"error": str(e)})
            return

        if result.get("error"):
            await state.set_state(PipelineStates.idle)
            await message.answer(f"{header}Parse error: {result['error']}\n\nPlease try again.")
            audit.log_entry("parse_error", {"error": result["error"]})
            return

        # Store pipeline result in FSM data
        parsed_race = result.get("parsed_race")
        await state.update_data(
            pipeline_result=result,
            parsed_race_json=parsed_race.model_dump_json() if parsed_race else None
        )

        # AUDIT-01: log parse result
        audit.log_entry("parse_complete", {
            "parsed_race": parsed_race.model_dump() if parsed_race else None,
            "overround_raw": result.get("overround_raw"),
            "overround_active": result.get("overround_active"),
            "ambiguous_fields": result.get("ambiguous_fields"),
        })

        # PIPELINE-01: progressive update — format and display
        summary = format_race_summary(result)

        # PIPELINE-02: Check for ambiguous data — if ambiguous_fields is non-empty,
        # ask clarifying question before showing confirmation
        ambiguous = result.get("ambiguous_fields", [])
        if ambiguous:
            await state.set_state(PipelineStates.awaiting_clarification)
            await state.update_data(ambiguous_fields=ambiguous)
            # Build a specific question about the ambiguous fields
            question_parts = []
            if "track" in ambiguous:
                question_parts.append("What is the track/venue name?")
            if "runner_count_mismatch" in ambiguous:
                question_parts.append(f"The paste mentions {parsed_race.runner_count} runners but only {len(parsed_race.runners)} were extracted. Is this correct, or should there be more?")
            if "missing_odds" in ambiguous:
                runners_no_odds = [r.name for r in parsed_race.runners if r.win_odds is None and r.status == "active"]
                question_parts.append(f"These runners are missing odds: {', '.join(runners_no_odds[:5])}. Are they scratched, or do they have odds?")
            if not question_parts:
                question_parts.append(f"Some data appears incomplete ({', '.join(ambiguous)}). Can you clarify or confirm this is correct?")

            audit.log_entry("clarification_asked", {"ambiguous_fields": ambiguous, "questions": question_parts})
            await message.answer(
                f"{header}{summary}\n\n"
                "<b>Clarification needed:</b>\n" +
                "\n".join(f"- {q}" for q in question_parts) +
                "\n\nPlease reply with the missing info, or type 'ok' to proceed as-is."
            )
            return

        # PARSE-04 / D-22: display summary and ask for confirmation
        await state.set_state(PipelineStates.awaiting_parse_confirm)
        await message.answer(
            f"{header}{summary}\n\n"
            "Please confirm this race data is correct:",
            reply_markup=confirm_parse_kb()
        )

    # INPUT-01: raw text paste
    @router.message(PipelineStates.idle, F.text)
    async def handle_paste(message: Message, state: FSMContext):
        # Skip if it looks like a command
        if message.text.startswith("/"):
            return
        await _run_parse_pipeline(message, state, message.text)

    # Also handle text when no state is set (first message)
    @router.message(F.text, ~F.text.startswith("/"))
    async def handle_paste_no_state(message: Message, state: FSMContext):
        current = await state.get_state()
        if current is None:
            await state.set_state(PipelineStates.idle)
            await _run_parse_pipeline(message, state, message.text)
        elif current != PipelineStates.idle.state:
            # PIPELINE-05: active pipeline warning
            settings = get_stake_settings()
            header = balance_header(settings.database_path)
            await message.answer(
                f"{header}"
                "A pipeline is already active. Use /cancel first."
            )

    # INPUT-02: .txt file upload
    @router.message(PipelineStates.idle, F.document)
    async def handle_document(message: Message, state: FSMContext, bot: Bot):
        doc = message.document
        if doc.mime_type not in ("text/plain",) and not (doc.file_name and doc.file_name.endswith(".txt")):
            await message.answer("Please send a .txt file with race data.")
            return
        buf = io.BytesIO()
        await bot.download(doc.file_id, destination=buf)
        raw_text = buf.getvalue().decode("utf-8")
        await _run_parse_pipeline(message, state, raw_text)

    # PIPELINE-02: Handle clarification response
    @router.message(PipelineStates.awaiting_clarification, F.text)
    async def handle_clarification(message: Message, state: FSMContext):
        """Handle user's response to clarifying question about ambiguous data."""
        settings = get_stake_settings()
        header = balance_header(settings.database_path)
        audit = AuditLogger()

        data = await state.get_data()
        user_response = message.text.strip()

        audit.log_entry("clarification_received", {"response": user_response})

        # If user says "ok" or similar, proceed with current parse as-is
        if user_response.lower() in ("ok", "okay", "yes", "proceed", "confirm", "y"):
            result = data.get("pipeline_result", {})
            summary = format_race_summary(result)
            await state.set_state(PipelineStates.awaiting_parse_confirm)
            await message.answer(
                f"{header}{summary}\n\n"
                "Please confirm this race data is correct:",
                reply_markup=confirm_parse_kb()
            )
            return

        # Otherwise, re-run pipeline with original text + user clarification appended
        original_input = data.get("raw_input", "")
        augmented_input = f"{original_input}\n\n[User clarification: {user_response}]"

        await state.set_state(PipelineStates.idle)
        await _run_parse_pipeline(message, state, augmented_input)

    # Handle bankroll input when waiting for manual entry
    @router.message(PipelineStates.awaiting_bankroll_input, F.text)
    async def handle_bankroll_input(message: Message, state: FSMContext):
        settings = get_stake_settings()
        header = balance_header(settings.database_path)
        audit = AuditLogger()

        text = message.text.strip()
        # Try to parse as number (user types "150" or "balance: 150")
        import re
        match = re.search(r"[\d.]+", text)
        if not match:
            await message.answer(
                f"{header}"
                "Please enter a valid USDT amount (e.g., 150):",
                reply_markup=bankroll_input_kb()
            )
            return

        try:
            amount = float(match.group())
            repo = BankrollRepository(db_path=settings.database_path)
            repo.set_balance(amount)
            stake_pct = repo.get_stake_pct()
            audit.log_entry("bankroll_set", {"balance": amount, "source": "manual_input"})

            await state.set_state(PipelineStates.idle)
            await message.answer(
                f"Balance set to {amount:.2f} USDT.\n"
                f"Stake size: {stake_pct*100:.1f}% ({amount * stake_pct:.2f} USDT per bet)\n"
                f"To adjust stake %: /stake <number>\n\n"
                "Pipeline complete. Paste new race data when ready."
            )
        except ValueError:
            await message.answer(
                f"{header}"
                "Invalid amount. Please enter a number (e.g., 150):",
                reply_markup=bankroll_input_kb()
            )
    ```
